CMAKE_MINIMUM_REQUIRED( VERSION 2.8.10 FATAL_ERROR )

PROJECT( CRIMILD )

OPTION( CRIMILD_ENABLE_TESTS "Would you like to enable unit tests?" OFF )

ENABLE_TESTING()

ADD_SUBDIRECTORY( src )
ADD_SUBDIRECTORY( examples )
ADD_SUBDIRECTORY( tests )
